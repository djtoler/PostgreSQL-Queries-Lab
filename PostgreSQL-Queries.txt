14. /* inner join */
select * 
from customers 
inner join orders
on customers.customer_id = orders.customer_id;

from customers 
right join orders
on customers.customer_id = orders.customer_id;

/* left join */
select * 
from customers 
left join orders
on customers.customer_id = orders.customer_id;
